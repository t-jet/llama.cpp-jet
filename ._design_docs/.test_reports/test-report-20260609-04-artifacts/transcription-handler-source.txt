            meta->chat_params,
            files);

            req,
            SERVER_TASK_TYPE_COMPLETION,
            body_parsed,
            files,
            TASK_RESPONSE_TYPE_OAI_ASR,
            body.contains("messages") ? &body.at("messages") : nullptr,
            "openai-transcription");
    };
